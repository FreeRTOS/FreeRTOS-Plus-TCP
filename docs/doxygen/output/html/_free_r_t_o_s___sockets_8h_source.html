221  static portINLINE unsigned int prvSocketValid( Socket_t xSocket )
222  {
223  unsigned int lReturnValue = pdFALSE;
224 
225  /*
226  * There are two values which can indicate an invalid socket:
227  * FREERTOS_INVALID_SOCKET and NULL. In order to compare against
228  * both values, the code cannot be compliant with rule 11.4,
229  * hence the Coverity suppression statement below.
230  */
231  /* coverity[misra_c_2012_rule_11_4_violation] */
232  if( ( xSocket != FREERTOS_INVALID_SOCKET ) && ( xSocket != NULL ) )
233  {
234  lReturnValue = pdTRUE;
235  }
236 
237  return lReturnValue;
238  }
