35  typedef enum eTCP_STATE
36  {
37  /* Comments about the TCP states are borrowed from the very useful
38  * Wiki page:
39  * http://en.wikipedia.org/wiki/Transmission_Control_Protocol */
40  eCLOSED = 0U, /* 0 (server + client) no connection state at all. */
41  eTCP_LISTEN, /* 1 (server) waiting for a connection request
42  from any remote TCP and port. */
43  eCONNECT_SYN, /* 2 (client) internal state: socket wants to send
44  a connect */
45  eSYN_FIRST, /* 3 (server) Just created, must ACK the SYN request. */
46  eSYN_RECEIVED, /* 4 (server) waiting for a confirming connection request
47  acknowledgement after having both received and sent a connection request. */
48  eESTABLISHED, /* 5 (server + client) an open connection, data received can be
49  delivered to the user. The normal state for the data transfer phase of the connection. */
50  eFIN_WAIT_1, /* 6 (server + client) waiting for a connection termination request from the remote TCP,
51  or an acknowledgement of the connection termination request previously sent. */
52  eFIN_WAIT_2, /* 7 (server + client) waiting for a connection termination request from the remote TCP. */
53  eCLOSE_WAIT, /* 8 (server + client) waiting for a connection termination request from the local user. */
54  eCLOSING, /* 9 (server + client) waiting for a connection termination request acknowledgement from the remote TCP. */
55  eLAST_ACK, /*10 (server + client) waiting for an acknowledgement of the connection termination request
56  previously sent to the remote TCP
57  (which includes an acknowledgement of its connection termination request). */
58  eTIME_WAIT, /*11 (either server or client) waiting for enough time to pass to be sure the remote TCP received the
59  acknowledgement of its connection termination request. [According to RFC 793 a connection can
60  stay in TIME-WAIT for a maximum of four minutes known as a MSL (maximum segment lifetime).] */
61  } eIPTCPState_t;
